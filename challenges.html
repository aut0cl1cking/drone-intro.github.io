Challenge 3: Aerial Photography

Description: Capture stunning images from above. The goal here is to combine stability in flight with photography skills.

Evaluation: The quality of the captured images is the main focus. Good images require smooth and stable flight and proper use of the camera.


import djitellopy
import time

tello = djitellopy.Tello()
tello.connect()

tello.takeoff()
tello.set_speed(100)

#CBM = crazy basic movement
def CBM():
    tello.move_up(50)
    tello.rotate_clockwise(45)
    tello.move_down(50)
    tello.rotate_clockwise(45)
    tello.move_left(50)
    tello.rotate_clockwise(45)
    tello.move_right(50)
    tello.rotate_clockwise(45)

#CAM = Crazy advanced movement
def CAM():
    tello.move_up(250)
    tello.move_forward(50)
    tello.move_left(50)
    tello.move_back(50)
    tello.move_right(50)
    tello.move_forward(50)
    tello.move_right()

#CFM = chill fun movement
def CFM():
    tello.rotate_clockwise(90)
    tello.rotate_counter_clockwise(90)
    tello.rotate_clockwise(90)
    tello.rotate_counter_clockwise(90)
    tello.move_back(100)
    tello.move_forward(200)
    tello.move_back()

CAM()
tello.land()
